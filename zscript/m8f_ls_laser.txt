class WM_PuffBase : Actor
{
  Default
  {
    Mass 0;
    Radius 1;
    Height 2;
    +NOBLOCKMAP;
    +NOGRAVITY;
    +BLOODLESSIMPACT;
    +PUFFONACTORS;
    +DONTSPLASH;
    +NOTRIGGER;
    +FORCEXYBILLBOARD;
    +NOTONAUTOMAP;
  }

  override void PostBeginPlay()
  {
    double s = CVar.GetCVar("m8f_ls_Scale", players[consolePlayer]).GetFloat();
    Scale.X  = s;
    Scale.Y  = s;
  }

  States
    {
    Spawn:
      TNT1 A 0;
      WDOB A 1 BRIGHT;
      Stop;
    }
}
